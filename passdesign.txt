typedef enum DrawType
{
    DRAW_DIRECT,
    DRAW_INDIRECT,
    DRAW_INDEXED,
    DRAW_INDEXED_INDIRECT,
    DRAW_INDEXED_INDIRECT_COUNT,
} DrawType;


typedef struct DrawCmd
{
    PipelineID pipeline;
    DrawType   type;

    union
    {
        struct
        {
            uint32_t vertexCount;
            uint32_t instanceCount;
            uint32_t firstVertex;
            uint32_t firstInstance;
        } direct;

        struct
        {
            VkBuffer     buffer;
            VkDeviceSize offset;
            uint32_t     drawCount;
            uint32_t     stride;
        } indirect;

        struct
        {
            uint32_t indexCount;
            uint32_t instanceCount;
            uint32_t firstIndex;
            int32_t  vertexOffset;
            uint32_t firstInstance;
        } indexed;

        struct
        {
            VkBuffer     buffer;
            VkDeviceSize offset;
            uint32_t     drawCount;
            uint32_t     stride;
        } indexed_indirect;

        struct
        {
            VkBuffer     buffer;
            VkDeviceSize offset;
            VkBuffer     countBuffer;
            VkDeviceSize countOffset;
            uint32_t     maxDrawCount;
            uint32_t     stride;
        } indexed_indirect_count;
    };

} DrawCmd;
#define MAX_PASS_RESOURCES 32
#define MAX_COLOR_ATTACHMENTS 8
typedef struct PassAttachments
{
    TextureHandle color[MAX_COLOR_ATTACHMENTS];
    uint32_t      color_count;

    TextureHandle depth;
    bool          has_depth;

} PassAttachments;


typedef enum
{
    PASS_GRAPHICS,
    PASS_COMPUTE,
} PassType;

typedef enum
{
    RESOURCE_TEXTURE,
    RESOURCE_BUFFER,
} ResourceType;

typedef enum
{
    ACCESS_READ,
    ACCESS_WRITE,
    ACCESS_READ_WRITE,
} ResourceAccess;

typedef enum
{
    USAGE_COLOR_ATTACHMENT,
    USAGE_DEPTH_ATTACHMENT,
    USAGE_SAMPLED,
    USAGE_STORAGE,
    USAGE_UNIFORM_BUFFER,
    USAGE_STORAGE_BUFFER,
} ResourceUsage;

typedef struct
{
    ResourceType type;
    union
    {
        TextureHandle texture_handle;
        BufferHandle  buffer_handle;
    };
    ResourceUsage  usage;
    ResourceAccess access;
} PassResource;


typedef struct GraphicsWork
{
    uint32_t draw_count;
    DrawCmd* draws;

} GraphicsWork;

typedef struct ComputeWork
{
    PipelineID pipeline;

    uint32_t group_x;
    uint32_t group_y;
    uint32_t group_z;

} ComputeWork;

typedef struct Pass
{
    const char* name;

    PassType type;

    PassAttachments attachments;

    uint32_t     resource_count;
    PassResource resources[MAX_PASS_RESOURCES];

    void*    push_data;
    uint32_t push_size;

    union
    {
        GraphicsWork graphics;
        ComputeWork  compute;
    };

} Pass;

static inline void execute_draw(VkCommandBuffer cmd, const DrawCmd* d)
{
    switch(d->type)
    {
        case DRAW_DIRECT:
            vkCmdDraw(cmd, d->direct.vertexCount, d->direct.instanceCount, d->direct.firstVertex, d->direct.firstInstance);
            break;

        case DRAW_INDIRECT:
            vkCmdDrawIndirect(cmd, d->indirect.buffer, d->indirect.offset, d->indirect.drawCount, d->indirect.stride);
            break;

        case DRAW_INDEXED:
            vkCmdDrawIndexed(cmd, d->indexed.indexCount, d->indexed.instanceCount, d->indexed.firstIndex,
                             d->indexed.vertexOffset, d->indexed.firstInstance);
            break;

        case DRAW_INDEXED_INDIRECT:
            vkCmdDrawIndexedIndirect(cmd, d->indexed_indirect.buffer, d->indexed_indirect.offset,
                                     d->indexed_indirect.drawCount, d->indexed_indirect.stride);
            break;

        case DRAW_INDEXED_INDIRECT_COUNT:
            vkCmdDrawIndexedIndirectCount(cmd, d->indexed_indirect_count.buffer, d->indexed_indirect_count.offset,
                                          d->indexed_indirect_count.countBuffer, d->indexed_indirect_count.countOffset,
                                          d->indexed_indirect_count.maxDrawCount, d->indexed_indirect_count.stride);
            break;
    }
}
